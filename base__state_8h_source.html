105  template <typename Ftype, typename Utype>
106  inline DEVICE Ftype uniform(const Utype in) const {
107  constexpr Ftype factor =
108  Ftype(1.) / (Ftype(~static_cast<Utype>(0)) + Ftype(1.));
109  constexpr Ftype halffactor = Ftype(0.5) * factor;
110  return Utype(in) * factor + halffactor;

182  template <typename T = float>
183  DEVICE inline T gamma(T alpha, T b) {
184  T d = alpha - T((1. / 3.));
185  T c = T(1.) / sqrt(9.f * d);
186  T v, x;
187  while (true) {
188  do {
189  x = randn<T>();
190  v = T(1.0) + c * x;
191  } while (v <= T(0.));
192  v = v * v * v;
193  T u = rand<T>();
194 
195  const T x2 = x * x;
196  if (u < 1.0f - 0.0331f * x2 * x2) return (d * v * b);
197 
198  if (log(u) < 0.5f * x2 + d * (1.0f - v + log(v))) return (d * v * b);
199  }
200  }
