26 inline DEVICE void round(const uint32_t (&key)[2], uint32_t (&ctr)[4]) {
27  uint32_t hi0;
28  uint32_t hi1;
29  uint32_t lo0 = mulhilo(PHILOX_M4x32_0, ctr[0], &hi0);
30  uint32_t lo1 = mulhilo(PHILOX_M4x32_1, ctr[2], &hi1);
31  ctr[0] = hi1 ^ ctr[1] ^ key[0];
32  ctr[1] = lo1;
33  ctr[2] = hi0 ^ ctr[3] ^ key[1];
34  ctr[3] = lo0;
