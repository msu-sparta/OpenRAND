40 constexpr uint32_t PHILOX_M4x32_0 = 0xD2511F53;
41 constexpr uint32_t PHILOX_M4x32_1 = 0xCD9E8D57;

43 constexpr uint32_t PHILOX_W32_1 = 0xBB67AE85;

45 inline DEVICE uint32_t mulhilo(uint32_t L, uint32_t R, uint32_t *hip) {
46  uint64_t product = static_cast<uint64_t>(L) * static_cast<uint64_t>(R);
47  *hip = product >> 32;
48  return static_cast<uint32_t>(product);
49 }

51 inline DEVICE void round(const uint32_t (&key)[2], uint32_t (&ctr)[4]) {
52  uint32_t hi0;
53  uint32_t hi1;
54  uint32_t lo0 = mulhilo(PHILOX_M4x32_0, ctr[0], &hi0);
55  uint32_t lo1 = mulhilo(PHILOX_M4x32_1, ctr[2], &hi1);
56  ctr[0] = hi1 ^ ctr[1] ^ key[0];
57  ctr[1] = lo1;
58  ctr[2] = hi0 ^ ctr[3] ^ key[1];
59  ctr[3] = lo0;
60 }

70 class Phillox : public BaseRNG<Phillox> {
71  public:
72  DEVICE Phillox(uint64_t seed, uint32_t ctr,

74  : seed_hi((uint32_t)(seed >> 32)),

81  DEVICE T draw() {

83 
84  static_assert(std::is_same_v<T, uint32_t> || std::is_same_v<T, uint64_t>);

89  return static_cast<uint64_t>(res);

104  private:
105  DEVICE void generate() {
106  uint32_t key[2] = {seed_hi, seed_lo};
107  // The counter takes one of the 4 values from internal counter, one from
108  // global seed, and one is what the user provided during instantiation.
109  // Another is left constant.
110 
111  // The internal counter helps to avoid forcing user to increment counter
112  // each time a number is generated.
113 
114  _out[0] = 0x12345;
115  _out[1] = _ctr;
116  _out[2] = initctr_hi;
117  _out[3] = initctr_lo;
118 
119  for (int r = 0; r < 10; r++) {
120  if (r > 0) {
121  key[0] += PHILOX_W32_0;
122  key[1] += PHILOX_W32_1;
123  }
124  round(key, _out);
125  }
126  _ctr++;
127  }
128 
129  // User provided seed and counter broken up, constant throughout
130  // the lifetime of the object
131  const uint32_t seed_hi, seed_lo;
132  const uint32_t initctr_hi, initctr_lo;
133  // private counter to keep track of numbers generated by this instance of rng
134  uint32_t _ctr = 0;
135  uint32_t _out[4];
136 }; // class Phillox
